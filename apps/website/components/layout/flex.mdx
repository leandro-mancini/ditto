## Uso

```jsx
import { Flex } from '@ditto/react';

<Flex>
  <div />
  <div />
</Flex>;
```

## Exemplos

### `direction`

Use a propriedade `direction` ou `flexDirection` para alterar a direção do flex.

```
<ExampleTabs name="flex-with-direction" />
```

```tsx
export const FlexWithDirection = () => {
  return (
    <Flex gap="4" direction="column">
      <Box height="10" />
      <Box height="10" />
      <Box height="10" />
    </Flex>
  );
};
```

### `align`

Use a propriedade `align` ou `alignItems` para alinhar os filhos ao longo do eixo transversal.

```
<ExampleTabs name="flex-with-align" />
```

```tsx
export const FlexWithAlign = () => {
  return (
    <Flex gap="4" align="center">
      <Box height="4" />
      <Box height="8" />
      <Box height="10" />
    </Flex>
  );
};
```

### `justify`

Use a propriedade `justify` ou `justifyContent` para alinhar os filhos ao longo do eixo principal.

```
<ExampleTabs name="flex-with-justify" />
```

```tsx
export const FlexWithJustify = () => {
  return (
    <Flex direction="column" gap="8">
      <Flex gap="4" justify="flex-start">
        <Box height="10" width="120px" />
        <Box height="10" width="120px">
          flex-start
        </Box>
        <Box height="10" width="120px" />
      </Flex>

      <Flex gap="4" justify="center">
        <Box height="10" width="120px" />
        <Box height="10" width="120px">
          center
        </Box>
        <Box height="10" width="120px" />
      </Flex>

      <Flex gap="4" justify="flex-end">
        <Box height="10" width="120px" />
        <Box height="10" width="120px">
          flex-end
        </Box>
        <Box height="10" width="120px" />
      </Flex>

      <Flex gap="4" justify="space-between">
        <Box height="10" width="120px" />
        <Box height="10" width="120px">
          space-between
        </Box>
        <Box height="10" width="120px" />
      </Flex>
    </Flex>
  );
};
```

### `order`

Use a propriedade `order` para alterar a ordem dos filhos.

```
<ExampleTabs name="flex-with-order" />
```

```tsx
export const FlexWithOrder = () => {
  return (
    <Flex gap="4">
      <Box height="10" order="1">
        1
      </Box>
      <Box height="10" order="3">
        2
      </Box>
      <Box height="10" order="2">
        3
      </Box>
    </Flex>
  );
};
```

### Margem Automática

Aplique margem a um item flex para afastá-lo de seus irmãos.

```
<ExampleTabs name="flex-with-auto-margin" />
```

```tsx
export const FlexWithAutoMargin = () => {
  return (
    <Flex gap="4" justify="space-between">
      <Box height="10" width="40" />
      <Box height="10" width="40" marginEnd="auto" />
      <Box height="10" width="40" />
    </Flex>
  );
};
```

### `wrap`

Use a propriedade `wrap` ou `flexWrap` para quebrar os filhos quando eles ultrapassarem o limite do elemento pai.

```
<ExampleTabs name="flex-with-wrap" />
```

```tsx
export const FlexWithWrap = () => {
  return (
    <Flex gap="4" wrap="wrap" maxW="500px">
      <Box height="10" width="200px" />
      <Box height="10" width="200px" />
      <Box height="10" width="200px" />
    </Flex>
  );
};
```
